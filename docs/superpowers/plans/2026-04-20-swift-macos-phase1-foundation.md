# PolicyDB macOS Client — Phase 1: Foundation

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Stand up the native macOS SwiftUI client's foundation — Xcode project, GRDB database layer, ported Python rules (currency, phone, email, UID, ref tag), and config reader — with parity-tested ports and no UI. The deliverable is a green test suite and a database connection that proves the architecture works before any UI gets built on top.

**Architecture:** A single-target macOS SwiftUI app (bundle ID `com.grantgreeson.policydb.mac`) inside `/Users/grantgreeson/Developer/policydb/PolicyDBMac/`. SPM dependencies for GRDB, Yams, and PhoneNumberKit. A `Rules/` module ports five Python functions from `src/policydb/utils.py` and `src/policydb/db.py` one-for-one. Parity is enforced by JSON fixtures generated by a Python script and replayed by Swift tests — if Python's output changes, the fixture regenerates and Swift must update to match.

**Tech Stack:** Swift 6.0, SwiftUI, macOS 26 Tahoe SDK (`@available(macOS 26.0, *)`), GRDB.swift 6.x (SQLite + WAL), Yams (YAML parsing), PhoneNumberKit (libphonenumber port), XCTest + swift-testing.

**Spec reference:** `docs/2026-04-20-swift-macos-frontend-design.md`

#!/usr/bin/env python3
"""Export Python rule outputs as JSON fixtures for Swift parity tests.

Regenerate whenever Python rule implementations change:

    ~/.policydb/venv/bin/python scripts/export_swift_fixtures.py

Writes to PolicyDBMac/PolicyDBMacTests/Fixtures/python-rule-outputs.json.
"""
from __future__ import annotations

import json
from pathlib import Path

from policydb.utils import (
    parse_currency_with_magnitude,
    format_phone,
    clean_email,
    build_ref_tag,
)
from policydb.db import generate_issue_uid  # used for shape verification only


REPO_ROOT = Path(__file__).resolve().parent.parent
OUT_PATH = REPO_ROOT / "PolicyDBMac" / "PolicyDBMacTests" / "Fixtures" / "python-rule-outputs.json"


CURRENCY_INPUTS = [
    "1m", "1M", "1.5m", "500k", "500K",
    "$2,000,000", "$1,234.56", "2500", "2500.00",
    "  $1,000  ", "", "abc",
    "1.2m", "0.5k", "2b",
]

PHONE_INPUTS = [
    "(555) 123-4567",
    "555-123-4567",
    "5551234567",
    "+1 555 123 4567",
    "+44 20 7946 0958",
    "",
    "not a phone",
]

EMAIL_INPUTS = [
    "Grant@Example.com",
    "  grant@example.com  ",
    "GRANT@EXAMPLE.COM",
    "grant+policy@example.com",
    "",
    "not an email",
    "user@",
]

REF_TAG_INPUTS = [
    {"cn_number": "CN1234567", "project_id": None, "policy_uid": None},
    {"cn_number": "CN1234567", "project_id": 3, "policy_uid": None},
    {"cn_number": "CN1234567", "project_id": 3, "policy_uid": "POL-042"},
    {"cn_number": None, "project_id": None, "policy_uid": "POL-001"},
    {"cn_number": "CN9999999", "project_id": 12, "policy_uid": "POL-500"},
]


def _safe_call(fn, value):
    """Run a function, return either the result or an error marker."""
    try:
        return {"input": value, "output": fn(value), "error": None}
    except Exception as exc:  # noqa: BLE001 — fixture wants to capture errors
        return {"input": value, "output": None, "error": f"{type(exc).__name__}: {exc}"}


def export_currency():
    return [_safe_call(parse_currency_with_magnitude, v) for v in CURRENCY_INPUTS]


def export_phone():
    return [_safe_call(format_phone, v) for v in PHONE_INPUTS]


def export_email():
    return [_safe_call(clean_email, v) for v in EMAIL_INPUTS]


def export_ref_tag():
    results = []
    for case in REF_TAG_INPUTS:
        try:
            output = build_ref_tag(
                cn_number=case["cn_number"],
                project_id=case["project_id"],
                policy_uid=case["policy_uid"],
            )
            results.append({"input": case, "output": output, "error": None})
        except Exception as exc:  # noqa: BLE001
            results.append({"input": case, "output": None, "error": f"{type(exc).__name__}: {exc}"})
    return results


def export_issue_uid_shape():
    """We can't fixture-test randomness, but we can verify shape."""
    samples = [generate_issue_uid() for _ in range(10)]
    return {
        "samples": samples,
        "length": 8,
        "charset": "0123456789ABCDEF",
    }


def main():
    fixture = {
        "parse_currency_with_magnitude": export_currency(),
        "format_phone": export_phone(),
        "clean_email": export_email(),
        "build_ref_tag": export_ref_tag(),
        "generate_issue_uid_shape": export_issue_uid_shape(),
    }
    OUT_PATH.parent.mkdir(parents=True, exist_ok=True)
    OUT_PATH.write_text(json.dumps(fixture, indent=2, default=str))
    print(f"Wrote {OUT_PATH}")


if __name__ == "__main__":
    main()
```

import Foundation

/// Parses currency strings matching PolicyDB's Python `parse_currency_with_magnitude`.
///
/// Accepts shorthand magnitudes (`1m`, `500k`, `2b`), dollar signs, commas, surrounding
/// whitespace. Returns `nil` for empty or unparseable input.
///
/// This must stay byte-identical to `src/policydb/utils.py::parse_currency_with_magnitude`.
/// Parity is enforced by `CurrencyParserTests`.
enum CurrencyParser {

    static func parse(_ raw: String) -> Double? {
        let trimmed = raw.trimmingCharacters(in: .whitespacesAndNewlines)
        guard !trimmed.isEmpty else { return nil }

        // Strip $ and commas
        var cleaned = trimmed
            .replacingOccurrences(of: "$", with: "")
            .replacingOccurrences(of: ",", with: "")

        // Detect trailing magnitude suffix
        var multiplier: Double = 1
        if let last = cleaned.last {
            switch last.lowercased() {
            case "k":
                multiplier = 1_000
                cleaned.removeLast()
            case "m":
                multiplier = 1_000_000
                cleaned.removeLast()
            case "b":
                multiplier = 1_000_000_000
                cleaned.removeLast()
            default:
                break
            }
            cleaned = cleaned.trimmingCharacters(in: .whitespaces)
        }

        guard let value = Double(cleaned) else { return nil }
        return value * multiplier
    }
}
```

import Foundation
import PhoneNumberKit

/// Formats phone numbers matching PolicyDB's Python `format_phone` (uses `phonenumbers`).
///
/// Returns `nil` when the input cannot be parsed; otherwise returns `ok=true` with
/// the E.164 and pretty-formatted variants. Parity with Python is enforced by
/// `PhoneFormatterTests`.
enum PhoneFormatter {

    struct Result: Codable, Equatable {
        let ok: Bool
        let formatted: String
        let e164: String
    }

    private static let util = PhoneNumberUtility()

    static func format(_ raw: String) -> Result? {
        let trimmed = raw.trimmingCharacters(in: .whitespacesAndNewlines)
        guard !trimmed.isEmpty else { return nil }

        do {
            let parsed = try util.parse(trimmed, withRegion: "US", ignoreType: true)
            return Result(
                ok: true,
                formatted: util.format(parsed, toType: .national),
                e164: util.format(parsed, toType: .e164)
            )
        } catch {
            return nil
        }
    }
}
```

import Foundation

/// Cleans email addresses matching PolicyDB's Python `clean_email`.
///
/// Trims whitespace, lowercases, validates a simple shape (has `@`, has dot after),
/// returns `nil` for invalid or empty input. Parity with Python is enforced by
/// `EmailCleanerTests`.
enum EmailCleaner {

    struct Result: Codable, Equatable {
        let ok: Bool
        let formatted: String
    }

    static func clean(_ raw: String) -> Result? {
        let trimmed = raw.trimmingCharacters(in: .whitespacesAndNewlines).lowercased()
        guard !trimmed.isEmpty else { return nil }

        // Minimal shape validation: one @, at least one dot after, no whitespace
        guard let atIdx = trimmed.firstIndex(of: "@") else { return nil }
        let afterAt = trimmed[trimmed.index(after: atIdx)...]
        guard afterAt.contains("."), !afterAt.contains(where: { $0.isWhitespace }) else {
            return nil
        }
        guard !trimmed[..<atIdx].isEmpty else { return nil }

        return Result(ok: true, formatted: trimmed)
    }
}
```

import Foundation

/// Builds hierarchical PolicyDB reference tags matching Python's `build_ref_tag`.
///
/// Hierarchy: `CN{cn}-L{project}-{policy_uid}`. Missing segments collapse —
/// e.g., a client-only record returns `CN{cn}`; a standalone policy returns
/// just `{policy_uid}`. Parity with Python is enforced by `RefTagBuilderTests`.
enum RefTagBuilder {

    static func build(cnNumber: String?, projectID: Int?, policyUID: String?) -> String? {
        var parts: [String] = []

        if let cn = cnNumber, !cn.isEmpty {
            parts.append(cn.hasPrefix("CN") ? cn : "CN\(cn)")
        }
        if let pid = projectID {
            parts.append("L\(pid)")
        }
        if let uid = policyUID, !uid.isEmpty {
            parts.append(uid)
        }

        return parts.isEmpty ? nil : parts.joined(separator: "-")
    }
}

import Foundation
import GRDB

/// Owns the PolicyDB SQLite connection. Runs as a coexisting reader/writer
/// alongside the Python webapp — Python is the schema owner, so this type
/// never runs migrations or creates tables.
///
/// Use `reader` for SELECTs and `writer` for INSERT/UPDATE/DELETE. Each write
/// runs as a short transaction — do not hold writes open across UI interactions
/// (see spec §5.5 Concurrent Writer Handling).
struct DatabaseManager {

    let dbPool: DatabasePool

    var reader: DatabaseReader { dbPool }
    var writer: DatabaseWriter { dbPool }

    init(path: String) throws {
        guard FileManager.default.fileExists(atPath: path) else {
            throw DatabaseManagerError.fileMissing(path)
        }
        var config = Configuration()
        config.busyMode = .timeout(5)
        self.dbPool = try DatabasePool(path: path, configuration: config)

        // Ensure WAL is on (Python webapp sets this, but we enforce for safety)
        try dbPool.write { db in
            try db.execute(sql: "PRAGMA journal_mode = WAL")
        }
    }

    static func `default`() throws -> DatabaseManager {
        let home = FileManager.default.homeDirectoryForCurrentUser
        let path = home.appendingPathComponent(".policydb/policydb.sqlite").path
        return try DatabaseManager(path: path)
    }

    func currentSchemaVersion() throws -> Int {
        try dbPool.read { db in
            try Int.fetchOne(db, sql: """
                SELECT MAX(version) FROM schema_version
            """) ?? 0
        }
    }
}

enum DatabaseManagerError: Error, CustomStringConvertible {
    case fileMissing(String)

    var description: String {
        switch self {
        case .fileMissing(let path):
            return "SQLite database not found at \(path). Is the Python webapp set up?"
        }
    }
}
